// Числа Фибоначчи
fun fibonacci(n) {
    if (n <= 1) {
        return n;
    }
    return fibonacci(n - 1) + fibonacci(n - 2);
}

print "Fibonacci sequence:";
let fib1 = fibonacci(1);
let fib2 = fibonacci(2);
let fib3 = fibonacci(3);
let fib4 = fibonacci(4);
let fib5 = fibonacci(5);
let fib6 = fibonacci(6);

print "F(1) = " + fib1;
print "F(2) = " + fib2;
print "F(3) = " + fib3;
print "F(4) = " + fib4;
print "F(5) = " + fib5;
print "F(6) = " + fib6;

// Сумма массива (эмулируем массив)
fun sumArray(n) {
    if (n <= 0) {
        return 0;
    }
    return n + sumArray(n - 1);
}

print "Sum of numbers from 1 to 5:";
print sumArray(5);

// Вложенные функции
fun outerFunction(x) {
    fun innerFunction(y) {
        return x + y;
    }
    return innerFunction(10);
}

let outerResult = outerFunction(5);
print "outerFunction(5) = " + outerResult;

// Сложные вычисления
fun power(base, exponent) {
    if (exponent == 0) {
        return 1;
    }
    return base * power(base, exponent - 1);
}

print "2^5 =";
print power(2, 5);
print "3^3 =";
print power(3, 3);

// Комбинация разных возможностей
fun complexCalculation(a, b, c) {
    let sum = a + b;
    let product = sum * c;
    
    if (product > 100) {
        return "Large result: " + product;
    } else {
        return "Small result: " + product;
    }
}

print complexCalculation(10, 20, 3);
print complexCalculation(1, 2, 3);
