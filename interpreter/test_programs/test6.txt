// Тест математических операций
print "Testing math operations:";
print "2 + 3 * 4 =";
print 2 + 3 * 4; // Должно быть 14

print "(2 + 3) * 4 =";
print (2 + 3) * 4; // Должно быть 20

print "10 / 3 =";
print 10 / 3;

print fullName;
