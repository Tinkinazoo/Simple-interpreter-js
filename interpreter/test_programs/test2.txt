// Простые условия
let age = 18;
if (age >= 18) {
    print "Adult";
} else {
    print "Minor";
}

// Вложенные условия
let score = 85;
if (score >= 90) {
    print "Grade: A";
} else if (score >= 80) {
    print "Grade: B";
} else if (score >= 70) {
    print "Grade: C";
} else if (score >= 60) {
    print "Grade: D";
} else {
    print "Grade: F";
}

// Логические операторы
let hasTicket = true;
let hasMoney = false;

if (hasTicket and hasMoney) {
    print "You can go to cinema and buy popcorn";
} else if (hasTicket or hasMoney) {
    print "You can go to cinema OR buy popcorn";
} else {
    print "You need both ticket and money";
}

// Сравнения
let a = 10;
let b = 20;
if (a < b) {
    print "a is less than b";
}
if (a != b) {
    print "a is not equal to b";
}
